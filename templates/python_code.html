num1 = 1
num2 = 2
def calculate(num1, num2, operation='addition'):
    output = 0
    if operation == 'addition':
        output = num1+num2
    elif operation == 'subtraction':
        output = num1-num2
    elif operation == 'multiplication':
        output = num1*num2
    elif operation == 'division':
        if num2 == 0:
            output = 'z'
        else:
            output = num1/num2
    print(output)
calculate(num1, num2)
